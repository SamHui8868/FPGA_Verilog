module _7seg(a,f);
input[3:0] a;
output[7:0] f;
reg[7:0] f;
always @(a)
case(a)
4'd0:f=8'b01000000;
4'd1:f=8'b01111001;
4'd2:f=8'b00100100;
4'd3:f=8'b00110000;
4'd4:f=8'b00011001;
4'd5:f=8'b00010010;
4'd6:f=8'b00000010;
4'd7:f=8'b01111000;
4'd8:f=8'b00000000;
4'd9:f=8'b00011000;
4'd10:f=8'b00001000;
4'd11:f=8'b00000011;
4'd12:f=8'b00100111;
4'd13:f=8'b00100001;
4'd14:f=8'b00000110;
4'd15:f=8'b00001110;
default f=8'b10000000;
endcase
endmodule
